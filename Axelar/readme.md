```
sudo tee /etc/systemd/system/axelard.service > /dev/null <<EOF
[Unit]
Description=axelar
After=network-online.target

[Service]
User=$USER
ExecStart=$(which axelard) start
Restart=always
RestartSec=3
LimitNOFILE=65535

[Install]
WantedBy=multi-user.target
EOF
```
